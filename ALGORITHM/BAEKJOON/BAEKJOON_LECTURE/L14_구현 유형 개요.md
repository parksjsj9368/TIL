**단계 4 : 코드 작성**

<br>

```python
n = int(input())
plans = list(map(str, input().split()))

# L,R,U,D에 따른 이동 방향
dx = [0,0,-1,1]
dy = [-1,1,0,0]
move_types = ['L','R','U','D']

# 현재 위치
x,y = 1,1

# 이동 계획을 하나씩 확인하기
for plan in plans :
    # 이동 후 좌표 구하기
    for i in range(len(move_types)) :
        if plan == move_types[i] :
            nx = x + dx[i]
            ny = y + dy[i]

    # 공간 벗어나는 경우 무시
    if (nx<1) or (ny<1) or (nx>n) or (ny>n) :
        continue
        
    # 이동 수행
    x,y = nx,ny
    
print(x,y)
```
